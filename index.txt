<script>
    let aiActive = false;
    let recording = false;
    let lastCommand = "";
    let isRunning = false; 

    // --- 1. Joystick Setup ---
    // CHANGED: We use a smaller 'size' (e.g., 100) so it fits inside the responsive CSS box.
    var manager = nipplejs.create({
        zone: document.getElementById('joystick-zone'),
        mode: 'static',
        position: {left: '50%', top: '50%'},
        color: 'white', /* Changed nipple color to white for contrast */
        size: 100 
    });

    // Event: Move
    manager.on('move', function (evt, data) {
        if (aiActive) return;
        
        document.getElementById('ai-btn').classList.add('disabled-control');
        document.getElementById('run-btn').classList.add('disabled-control');

        let deg = data.angle.degree;
        let cmd = "";

        if (deg >= 67.5 && deg < 112.5) cmd = "up";
        else if (deg >= 112.5 && deg < 157.5) cmd = "up_left";
        else if (deg >= 157.5 && deg < 202.5) cmd = "left";
        else if (deg >= 202.5 && deg < 247.5) cmd = "down_left";
        else if (deg >= 247.5 && deg < 292.5) cmd = "down";
        else if (deg >= 292.5 && deg < 337.5) cmd = "down_right";
        else if (deg >= 22.5 && deg < 67.5) cmd = "up_right";
        else cmd = "right";

        if (cmd !== lastCommand) {
			lastCommand = cmd;
            sendCommand(cmd);
        }
    });

    // Event: Release (Stop)
    manager.on('end', function (evt, data) {
        if (aiActive || isRunning) return;
        
        document.getElementById('ai-btn').classList.remove('disabled-control');
        document.getElementById('run-btn').classList.remove('disabled-control');
        
        console.log("Joystick released");
        lastCommand = ""; 
        sendCommand("stop"); 
    });

    // --- 2. Logic Functions ---

    function sendCommand(command) {
        if(command === 'stop') {
            document.getElementById('status-log').innerText = "Status: Stopped";
        } else if (command !== 'accelerate' && command !== 'decelerate') {
            document.getElementById('status-log').innerText = "Cmd: " + command;
        }

        fetch('/move', {
            method: 'POST',
            headers: {'Content-Type': 'application/json'},
            body: JSON.stringify({command: command})
        })
        .then(res => res.json())
        .then(data => {
            if(data.speed) {
                 document.getElementById('status-log').innerText = "Speed Delay: " + data.speed + "s";
            }
        })
        .catch(e => console.error(e));
    }

    function toggleRun() {
        fetch('/move', {
            method: 'POST',
            headers: {'Content-Type': 'application/json'},
            body: JSON.stringify({command: 'toggle_run'})
        })
        .then(r => r.json())
        .then(data => {
            const btn = document.getElementById('run-btn');
            const aiBtn = document.getElementById('ai-btn');
            const joystick = document.getElementById('joystick-zone');
            
            isRunning = data.forward_moving;
            
            if (isRunning) {
                btn.innerHTML = '<span class="btn-icon">&#9632;</span> STOP';
                btn.classList.add('running');
                
                aiBtn.classList.add('disabled-control');
                joystick.classList.add('disabled-control');
                
            } else {
                btn.innerHTML = '<span class="btn-icon">&#9679;</span> RUN';
                btn.classList.remove('running');
                
                aiBtn.classList.remove('disabled-control');
				joystick.classList.remove('disabled-control');
            }
        });
    }

    function toggleAI() {
		fetch('/toggle_ai', {method: 'POST'})
            .then(response => response.json())
            .then(data => {
                aiActive = data.ai_tracking_active;
                const btn = document.getElementById('ai-btn');
                const videoRow = document.getElementById('video-row');
                const runBtn = document.getElementById('run-btn');
                const joystick = document.getElementById('joystick-zone');
                btn.classList.toggle('active', aiActive);
                
                
                if (aiActive) {
                    btn.textContent = "Stop AI";
                    
                    runBtn.classList.add('disabled-control');
                    joystick.classList.add('disabled-control');
                    
                    // Updated to use responsive CSS on images with strict max-height
                    videoRow.innerHTML = `
                        <div class="video-box"><h3>Original</h3><img src="{{ url_for('video_feed_original') }}" style="max-height:25vh"></div>
                        <div class="video-box"><h3>Detection</h3><img src="{{ url_for('video_feed_detection') }}" style="max-height:25vh"></div>
                        <div class="video-box"><h3>Mask</h3><img src="{{ url_for('video_feed_mask') }}" style="max-height:25vh"></div>
                    `;
                } else {
                    btn.textContent = "AI Tracking";
                    
                    runBtn.classList.remove('disabled-control');
                    joystick.classList.remove('disabled-control');
                    
                    videoRow.innerHTML = `
                        <div class="video-box"><h3>Original</h3><img src="{{ url_for('video_feed_original') }}"></div>
                    `;
                }
            });
    }
    
    function toggleRecord() {
		fetch('/record', {method: 'POST'})
            .then(response => response.json())
            .then(data => {
            recording = (data.status === 'started');
            const btn = document.getElementById('record-btn');
            if (recording) {
                btn.textContent = "Stop Record";
                btn.classList.add('recording');
            } else {
                btn.textContent = "Start Record";
                btn.classList.remove('recording');
            }
        });
    }
</script>
